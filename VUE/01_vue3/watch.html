                        # 组合式 api 中用法

                        # props 监听

                        watch(
  () => props.visible,
  (visible) => {
    if (visible) {
      init()
    }
  },
  {
    immediate: true,
  },
)

                        # array ref 监听

                        const list = ref<ItemType[]>([])

watch(list.value, (list, preList) => {

})

// 注意：直接监听 list 无效
