## sql

```sql
CREATE DATABASE IF NOT EXISTS LoveLyrics;
USE LoveLyrics;
CREATE TABLE Lyrics
(
  id int NOT NULL AUTO_INCREMENT,
  name char(50) NOT NULL,
  author char(50) NULL,
  grade int NULL,
  status int NULL,
  content text NULL,
  PRIMARY KEY(id)
);
```
