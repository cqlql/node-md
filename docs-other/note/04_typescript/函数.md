## 重载

方式 1：直接声明

方式 2： 单独声明

```ts
interface GetAddress {
  // 1 获取所有
  (): Promise<string[]>
  // 2 根据id 获取
  (data: { id: string }): Promise<string>
}

export const getAddress: GetAddress = (data) => {
  return post('/applets_address/get', data)
}

// 1 获取所有
getAddress().then((data) => {
  list.value = data.list
})

// 2 根据id 获取
getAddress({ id: '1' }).then((data) => {
  list.value = data
})
```
