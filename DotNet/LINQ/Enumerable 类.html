                                                                                                                                # Range 生成指定范围的整数序列

                                                                                                                                # 基础

                                                                                                                                IEnumerable<int> nums = Enumerable.Range(1, 3);

foreach (int num in nums) {
  // 依次输出：1, 2, 3
  Console.WriteLine(num);
}

                                                                                                                                # 配合 Select

                                                                                                                                有点类似 js 的 map

                                                                                                                                IEnumerable<int> nums = Enumerable.Range(1, 5).Select(v => v+1);

foreach (int num in nums) {
  Console.WriteLine(num);
}
