```cs
  [HttpPost("Add")]
  public APIResult<object> Add(User user)
  {
    // mark product as added in change tracking
    _db.Add(user);
    // save tracked changes to database 
    int affected = _db.SaveChanges();

    if (affected == 1)
    {
      return APIResult.Ok();
    }

    return APIResult.Error<object>("新增失败");
  }
```
