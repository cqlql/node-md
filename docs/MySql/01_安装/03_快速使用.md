如果是 docker ，则先要进入 mysql 容器环境

```sh

# 创建 mysql 容器
docker run --name some-mysql -p 3306:3306 -e MYSQL_ROOT_PASSWORD=123123 -d mysql:latest

# 列出 mysql 容器 id
# 此命令会列出正在运行的容器ID
docker ps

# 进入容器，相当于是一个linux系统
# 其中 289cc00dc5ed 为 mysql 容器 id
docker exec -it 289cc00dc5ed bash

```

进入 mysql

```sh
mysql -uroot -p
# 接下来输入root用戶密码
```

创建用户，数据库表，权限

```sql
create user 'joly'@'%' identified by "123123";
-- `identified by` 不带则不给密码
-- `%` 表示所有 host 都能访问，可以指定为 `localhost` 只能本地访问

-- 创建 db_user 数据库
CREATE DATABASE IF NOT EXISTS db_user;


-- 给 joly 分配 db_user 数据库的所有权限
grant all on db_user.* to 'joly'@'%';


-- 创建用户表
CREATE TABLE db_user.user (
  id INT NOT NULL AUTO_INCREMENT,
  username CHAR(45) NOT NULL,
  password CHAR(45) NOT NULL,
  nickname CHAR(45) NOT NULL,
  PRIMARY KEY (id)
);

-- 插入数据
INSERT INTO db_user.user (username,password,nickname)
  VALUES ('joly','123123','张三');

```
